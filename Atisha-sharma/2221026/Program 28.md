## Program 28: Write a program to display age of a family
#include<stdio.h>

int main()

{

float age1;

double age2;

printf("Please enter your age:");

scanf("%f",&age1);

printf("Enter your child age:");

scanf("%lf",&age2);

printf("\nage1=%f",age1);

printf("\nage2=%lf",age2);

return 0;
}

Output :

Please enter your age:43
  
Enter your child age:15
  
age1=43.000000
  
age2=15.000000
