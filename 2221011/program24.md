## Write a code using float|double datatype
#include <stdio.h>

int main()

{

    float age1;
    
    double age2;
    
    printf("Please enter your age:");
    
    scanf("%f",&age1);
    
    printf("Enter your child age:");
    
    scanf("%lf",age2);
    
    printf("age1=%f",age1);
    
    printf("age2=%lf",age2);
    
return 0;

}

**Output:Please enter your age:45 Enter your child age:13age1=45age2=13**
