## Program 28: Write a program to use palindrome string
```
#include <stdio.h>
#include <string.h>
int isPalindrome(char *str, int l, int r)
{
if(NULL == str || l < 0 || r < 0)
{
return 0;
}
if(l >= r)
{
return 1;
}
if(str[l] == str[r])
{
return isPalindrome(str, l + 1, r - 1);
}
return 0;
}
int main()
{
char s[] = {"abbba"} ;

}

}
```
