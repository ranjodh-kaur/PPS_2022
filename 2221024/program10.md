## Write a code for Bitwise Operters

#include <stdio.h>

int main() {

    int x=0,y=10;
    
    printf("The bitwise operators x&y = %d",x&y);
    
    printf("\nThe bitwise operators x|y = %d",x|y);
    
    printf("\nThe bitwise operaters x^y = %d",x^y);
    
    return 0;
    
}

**Output =The bitwise operators x&y = 0
          The bitwise operators x|y = 10
          The bitwise operaters x^y = 10**
